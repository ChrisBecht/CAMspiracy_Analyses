merge data

## remove network indicators
# colnames(t2_questionnaireCAMs[, c(48:79)])
t2_questionnaire <- t2_questionnaireCAMs[,-c(48:79)]



if (all(t2_questionnaire$PROLIFIC_PID == networkIndicators$participantCAM)) {
  print("all t2 data can be matched")
  t2_questionnaireCAMs <-
    cbind(t2_questionnaire, networkIndicators)
  
  
  
  ## add duration CAM
  t2_questionnaireCAMs$CAM_duration <- NA
  
  h = 1
  for (c in t2_questionnaireCAMs$participantCAM) {
    # print(c)
    
    tmp_diffConcepts <-
      CAMfiles[[1]][CAMfiles[[1]]$participantCAM == c, "dateConceptCreated"] -
      CAMfiles[[1]][CAMfiles[[1]]$participantCAM == c, "dateCAMcreated"]
    tmp_diffConnectors <-
      CAMfiles[[2]][CAMfiles[[2]]$participantCAM == c, "dateConnectorCreated"] -
      CAMfiles[[2]][CAMfiles[[2]]$participantCAM == c, "dateCAMcreated"]
    
    t2_questionnaireCAMs$CAM_duration[h] <-  max(c(tmp_diffConcepts, tmp_diffConnectors))
    h = h + 1
  }
  
  ## in minutes
  t2_questionnaireCAMs$CAM_duration <- t2_questionnaireCAMs$CAM_duration / 60
  
}
